#!/bin/bash

# Script para verificar permisos de usuario en el sistema
# Autor: Script de verificación de permisos
# Descripción: Muestra información sobre permisos de root, sudo y grupos del usuario

# Archivo de salida
OUTPUT_FILE="/tmp/permisos_$(date +%Y%m%d_%H%M%S).txt"

# Función para escribir en pantalla y archivo
log_output() {
    echo "$1" | tee -a "$OUTPUT_FILE"
}

# Limpiar archivo si existe
> "$OUTPUT_FILE"

log_output "╔════════════════════════════════════════════════════════════════╗"
log_output "║        VERIFICADOR DE PERMISOS DE USUARIO DEL SISTEMA         ║"
log_output "╚════════════════════════════════════════════════════════════════╝"
log_output ""

# Información del usuario actual
log_output "👤 INFORMACIÓN DEL USUARIO:"
log_output "─────────────────────────────────────────────────────────────────"
log_output "  Usuario: $(whoami)"
log_output "  UID: $(id -u)"
log_output "  GID: $(id -g)"
log_output "  Grupos: $(id -G)"
log_output ""

# Verificar si es root
log_output "🔐 ESTADO DE ROOT:"
log_output "─────────────────────────────────────────────────────────────────"
if [ "$(id -u)" -eq 0 ]; then
    log_output "  ✅ Eres ROOT (UID=0)"
    log_output "  Tienes acceso total al sistema"
else
    log_output "  ❌ NO eres root"
    log_output "  UID actual: $(id -u) (root sería 0)"
fi
log_output ""

# Verificar acceso a sudo
log_output "🔑 PERMISOS SUDO:"
log_output "─────────────────────────────────────────────────────────────────"
if sudo -n true 2>/dev/null; then
    log_output "  ✅ Puedes ejecutar sudo SIN contraseña"
    log_output "  (sudoers configurado sin NOPASSWD)"
else
    if sudo -l 2>/dev/null | grep -q "sudo"; then
        log_output "  ✅ Puedes ejecutar sudo CON contraseña"
        log_output "  (estás en el grupo sudoers)"
    else
        log_output "  ❌ NO tienes acceso a sudo"
    fi
fi
log_output ""

# Mostrar línea sudoers del usuario (si aplica)
log_output "📋 CONFIGURACIÓN SUDOERS:"
log_output "─────────────────────────────────────────────────────────────────"
SUDO_CONFIG=$(sudo -l 2>&1)
if echo "$SUDO_CONFIG" | grep -q "User $(whoami) may run"; then
    log_output "$(echo "$SUDO_CONFIG" | grep "may run")"
else
    if echo "$SUDO_CONFIG" | grep -q "not allowed"; then
        log_output "  ❌ Usuario no está en sudoers"
    fi
fi
log_output ""

# Verificar pertenencia a grupos importantes
log_output "👥 GRUPOS DEL USUARIO:"
log_output "─────────────────────────────────────────────────────────────────"
GRUPOS=$(id -Gn)
log_output "  Grupos: $GRUPOS"
log_output ""

# Verificar grupos especiales
log_output "  Pertenencia a grupos importantes:"
for grupo in sudo wheel docker adm systemd-journal; do
    if id -G $(whoami) | grep -qw "$(getent group $grupo | cut -d: -f3)"; then
        log_output "    ✅ $grupo"
    else
        log_output "    ❌ $grupo"
    fi
done
log_output ""

# Verificar permisos en directorios críticos
log_output "📁 PERMISOS EN DIRECTORIOS CRÍTICOS:"
log_output "─────────────────────────────────────────────────────────────────"
log_output "  Permiso lectura /root:"
if [ -r /root ]; then
    log_output "    ✅ Puedes leer /root"
else
    log_output "    ❌ No puedes leer /root"
fi

log_output "  Permiso escritura /etc:"
if [ -w /etc ]; then
    log_output "    ✅ Puedes escribir en /etc"
else
    log_output "    ❌ No puedes escribir en /etc"
fi

log_output "  Permiso lectura /var/log:"
if [ -r /var/log ]; then
    log_output "    ✅ Puedes leer /var/log"
else
    log_output "    ❌ No puedes leer /var/log"
fi
log_output ""

# Resumen de capacidades
log_output "📊 RESUMEN DE CAPACIDADES:"
log_output "─────────────────────────────────────────────────────────────────"
if [ "$(id -u)" -eq 0 ]; then
    log_output "  🟢 NIVEL: Administrador (ROOT)"
    log_output "  Tienes control total del sistema"
elif sudo -n true 2>/dev/null; then
    log_output "  🟡 NIVEL: Administrador (sudo sin contraseña)"
    log_output "  Puedes ejecutar comandos como root sin ingresar contraseña"
elif sudo -l 2>/dev/null | grep -q "sudo"; then
    log_output "  🟡 NIVEL: Usuario con privilegios elevados (sudo)"
    log_output "  Puedes ejecutar comandos como root con contraseña"
else
    log_output "  🔴 NIVEL: Usuario normal"
    log_output "  Solo tienes permisos estándar de usuario"
fi
log_output ""

log_output "═════════════════════════════════════════════════════════════════"
log_output ""
log_output "📄 Resultados guardados en: $OUTPUT_FILE"
