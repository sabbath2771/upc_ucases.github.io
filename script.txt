#!/bin/bash

# Script to run GNOME Calculator
# This script opens the GNOME calculator

# Check if gnome-calculator is installed
if command -v gnome-calculator &> /dev/null; then
    echo "Starting GNOME Calculator..."
    gnome-calculator &
else
    echo "Error: gnome-calculator is not installed"
    echo "Installing gnome-calculator..."
    
    # Try to install based on the available package manager
    if command -v apt &> /dev/null; then
        sudo apt update && sudo apt install -y gnome-calculator
    elif command -v dnf &> /dev/null; then
        sudo dnf install -y gnome-calculator
    elif command -v pacman &> /dev/null; then
        sudo pacman -S gnome-calculator
    fi
    
    # Run the calculator
    gnome-calculator &
fi

echo "GNOME Calculator is running..."
exit 0
